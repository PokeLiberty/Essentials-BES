def refresh
  bitmap=self.bitmap
  bitmap.fill_rect(self.x,self.y,self.width,self.height,Color.new(0,0,0,0))
  size=bitmap.text_size(self.label).width
  shadowtext(bitmap,self.x+4,self.y,size,self.height,self.label,@disabled)
end
