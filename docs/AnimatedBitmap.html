def initialize(file,hue=0)
  raise "filename is nil" if file==nil
  if file[/^\[(\d+)\]/]   # Starts with 1 or more digits in brackets
    @bitmap=PngAnimatedBitmap.new(file,hue)
  else
    @bitmap=GifBitmap.new(file,hue)
  end
end
