def pbStartScreen(pokemon)
  moves=pbGetRelearnableMoves(pokemon)
  @scene.pbStartScene(pokemon,moves)
  loop do
    move=@scene.pbChooseMove
    if move<=0
      if @scene.pbConfirm(
        _INTL("¿Prefieres que {1} no aprenda un movimiento nuevo?",pokemon.name))
        @scene.pbEndScene
        return false
      end
    else
      if @scene.pbConfirm(_INTL("¿Enseñar {1}?",PBMoves.getName(move)))
        if pbLearnMove(pokemon,move)
          @scene.pbEndScene
          return true
        end
      end
    end
  end
end
