def update
  @frame+=1
  @updating=true
  @spriteYExtra=0
  # Pok√©mon sprite agit√°ndose mientras es seleccionado el Pok√©mon
  if ((@frame/10).floor&1)==1 && @selected==1     # Cuando se escoge un comando para este Pok√©mon
    @spriteYExtra=2
  end
  self.x=@spriteX
  self.y=@spriteY
  self.visible=@spriteVisible
  # Pok√©mon sprite parpadeo cuando es apuntado o da√±ado
  if @spriteVisible && @selected==2       # Cuando es apuntado
    self.visible=(@frame%20<14)
  elsif @spriteVisible && @selected==3    # Cuando es da√±ado
    self.visible=(@frame%10<7)
  end
  if @_iconbitmap
    @_iconbitmap.update
    self.bitmap=@_iconbitmap.bitmap
  end
  @updating=false
end
