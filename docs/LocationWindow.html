def update
  return if @window.disposed?
  @window.update
  if $game_temp.message_window_showing ||
     @currentmap!=$game_map.map_id
    @window.dispose
    return
  end
  if @frames>80
    @window.y-=4
    @window.dispose if @window.y+@window.height<0
  else
    @window.y+=4 if @window.y<0
    @frames+=1
  end
end
