def initialize
  @items=[]
  # Start storage with a Potion
  if hasConst?(PBItems,:POTION)
    ItemStorageHelper.pbStoreItem(
       @items,MAXSIZE,MAXPERSLOT,getConst(PBItems,:POTION),1)
  end
end

def getCount(index)
  if index<0 || index>=@items.length
    return 0
  else
    return @items[index][1]
  end
end
