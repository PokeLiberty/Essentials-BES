def self.targetsOneOpponent?(move)
  return move.target==PBTargets::SingleNonUser ||
         move.target==PBTargets::RandomOpposing ||
         move.target==PBTargets::SingleOpposing ||
         move.target==PBTargets::OppositeOpposing
end
