def ensurePoint(j)
  if !@points[j] || @points[j].disposed?
    @points[j]=BitmapSprite.new(8,8,@viewport)
    @points[j].bitmap.fill_rect(0,0,8,8,Color.new(0,0,0))
  end
  @points[j].tone=(@strength==2) ? Tone.new(232,232,248) : 
     Tone.new(16,16,232)
  @points[j].visible=(@strength!=0)
end

def update
  for point in @points
    point.update
    point.visible=false
  end
  j=0
  i=0; while i<@distance
    o=(i+@offset)%@distance
    if o>=0 && o<@distance
      ensurePoint(j)
      pt=calcPoint(@x,@y,o,@angle)
      @points[j].x=pt[0]
      @points[j].y=pt[1]
      j+=1
    end
    i+=(@strength==2) ? 16 : 32
  end
  @offset+=(@strength==2) ? 3 : 2
  @offset%=@distance
end
