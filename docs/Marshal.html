def load(port,*arg)
  if @@neverload
    if port.is_a?(IO)
      return port.read
    else
      return port
    end
  end
  oldpos=port.pos if port.is_a?(IO)
  begin
    oldload(port,*arg)
  rescue
    p [$!.class,$!.message,$!.backtrace]
    if port.is_a?(IO)
      port.pos=oldpos
      return port.read
    else
      return port
    end
  end
end
