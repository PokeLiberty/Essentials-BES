def shadowtext(bitmap,x,y,w,h,t,disabled=false,align=0)
  width=bitmap.text_size(t).width
  if align==2
    x+=(w-width)
  elsif align==1
    x+=(w/2)-(width/2)
  end
  pbDrawShadowText(bitmap,x,y,w,h,t,
     disabled ? Color.new(26*8,26*8,25*8) : Color.new(12*8,12*8,12*8),
     Color.new(26*8,26*8,25*8))
end
