def self.getDataFromResult(result)
  datasize=CopyDataString.call(result,"",0)
  ret=nil
  if datasize!=0
    data="0"*datasize
    CopyDataString.call(result,data,datasize)
    ret=data.unpack("V*")
  end
  FreeDataString.call(result)
  return ret
end
