def getCurrentStorage
  if !$game_map
    raise _INTL("El jugador no está en el mapa, por lo que no se puede determinar la región.")
  end
  if @lastmap!=$game_map.map_id
    @rgnmap=pbGetCurrentRegion # may access file IO, so caching result
    @lastmap=$game_map.map_id
  end
  if @rgnmap<0
    raise _INTL("El mapa actual no pertenece a ninguna región. Por favor, configura los metadatos MapPosition para este mapa.")
  end
  if !@storages[@rgnmap]
    @storages[@rgnmap]=PokemonStorage.new
  end
  return @storages[@rgnmap]
end
