def initialize(file,savefile,prompt)
  @savefile=savefile
  @file=file
  if pbRgssExists?(@savefile)
    @ts=load_data(@savefile)
    if !@ts.changed? || prompt.call==true
      @data=Marshal.load(StringInput.new(@ts.data))
    else
      @ts=PokemonDataCopy.new(@file,@savefile)
      @data=load_data(@file)
    end
  else
    @ts=PokemonDataCopy.new(@file,@savefile)
    @data=load_data(@file)
  end
end
