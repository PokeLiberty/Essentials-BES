def initialize(width, height)
  @tileset_id = 1
  @width = width
  @height = height
  @autoplay_bgm = false
  @bgm = RPG::AudioFile.new
  @autoplay_bgs = false
  @bgs = RPG::AudioFile.new("", 80)
  @encounter_list = []
  @encounter_step = 30
  @data = Table.new(width, height, 3)
  @events = {}
end
