def update(matches)
  if matches.length == 0
    return
  end
  stake=0
  for i in 0...matches.length
    score=(match.score==-1) ? 0.5 : match.score
    e=(1+10.0**((@rating-match.opponentRating)/400.0))
    stake+=match.kValue*(score-e)
  end
  @rating+=stake
end
