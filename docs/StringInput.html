def gets
  raise IOError, 'closed stream' if @closed
  if idx = @string.index(?\n, @pos)
    idx += 1  # "\n".size
    line = @string[ @pos ... idx ]
    @pos = idx
    @pos += 1 if @pos == @string.size
  else
    line = @string[ @pos .. -1 ]
    @pos = @string.size + 1
  end
  @lineno += 1
  line
end

def read( len = nil )
  raise IOError, 'closed stream' if @closed
  if !len
    return nil if eof?
    rest = @string[@pos ... @string.size]
    @pos = @string.size + 1
    return rest
  end
  str = @string[@pos, len]
  @pos += len
  @pos += 1 if @pos == @string.size
  str
end

def seek( offset, whence=IO::SEEK_SET )
  raise IOError, 'closed stream' if @closed
  case whence
  when IO::SEEK_SET
    @pos=offset
  when IO::SEEK_CUR
    @pos+=offset
  when IO::SEEK_END
    @pos=@string.size - offset
  else
    raise ArgumentError, "unknown seek flag: #{whence}"
  end
  @pos = 0 if @pos < 0
  @pos = [@pos, @string.size + 1].min
  offset
end
