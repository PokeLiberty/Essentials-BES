def check(name,species)
  name=name.upcase
  return true if name==getName(species)
  if @@names.values.include?(name) 
    return false
  end
  for i in @@namesMaxValue..PBSpecies.maxValue
    if i!=species
      n=getName(i)
      return false if n==name
    end
  end
  return true
end

def getName(species)
  n=@@names[species]
  return n if n
  n=PBSpecies.getName(species)
  @@names[species]=n.upcase
  return n
end
