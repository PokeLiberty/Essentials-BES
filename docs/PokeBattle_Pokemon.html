  stats=[]
  pvalues=[100,100,100,100,100]

def changeHappiness(method)

  case method
  when "walking"

    gain=5 if @happiness<100
    luxury=true

    gain=4

  end

    self.shadowmoves[4+i]=self.moves[i].id
  end
  pbUpdateShadowMoves
end

      moves=[]

      relearned=0

      # Agrega X movimientos regulares

        end

    end
  end
