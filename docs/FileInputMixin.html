def fgetdw
  x=0
  ret=0
  each_byte do |i|
    break if !i
    ret|=(i<<x)
    x+=8
    break if x==32
  end
  return ret
end

def fgetsb
  ret=fgetb
  if (ret&0x80)!=0
    return ret-256
  else
    return ret
  end
end

def getLength(index)
  self.binmode
  self.pos=0
  offset=fgetdw>>3
  return 0 if index>=offset
  self.pos=index*8+4
  return fgetdw
end

def readName(index)
  self.binmode
  self.pos=0
  offset=fgetdw>>3
  return "" if index>=offset
  self.pos=index<<3
  offset=fgetdw
  length=fgetdw
  return "" if length==0
  self.pos=offset
  return read(length)
end
