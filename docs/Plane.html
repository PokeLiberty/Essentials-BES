def initialize(viewport=nil)
  _initialize_SpriteResizer(viewport)
  @resizedZoomX=1.0
  @resizedZoomY=1.0
  @resizedOx=0
  @resizedOy=0
  _zoomxeq_SpriteResizer(@resizedZoomX*$ResizeFactorMul/100)
  _zoomyeq_SpriteResizer(@resizedZoomY*$ResizeFactorMul/100)
end
