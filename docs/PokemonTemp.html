def pbGetBerryPlantData(item)
  if !@berryPlantData
    pbRgssOpen("Data/berryplants.dat","rb"){|f|
       @berryPlantData=Marshal.load(f)
    }
  end
  return @berryPlantData[item] if @berryPlantData && @berryPlantData[item]!=nil
  return [3,15,2,5] # Hours/stage, drying/hour, min yield, max yield
end

def pbOpenMoveData
  if !self.pokemonMoveData
    pbRgssOpen("Data/moves.dat","rb"){|f|
       self.pokemonMoveData=f.read
    }
  end
  if block_given?
    StringInput.open(self.pokemonMoveData) {|f| yield f }
  else
    return StringInput.open(self.pokemonMoveData)
  end
end
