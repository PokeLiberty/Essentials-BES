def refresh
  self.bitmap.clear
  for i in 0...5
    digit=(@score/(10**i))%10 # Least significant digit first
    self.bitmap.blt(14*(4-i),0,@numbers.bitmap,Rect.new(digit*14,0,14,22))
  end
end
