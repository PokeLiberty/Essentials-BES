def initialize(pokemon,viewport=nil)
  super(0,0,viewport)
  @release=Interpolator.new
  @startRelease=false
  @pokemon=pokemon
  if pokemon
    self.setBitmap(pbPokemonIconFile(pokemon))
  end
  self.src_rect=Rect.new(0,0,64,64)
end

def release
  self.ox=32
  self.oy=32
  self.x+=32
  self.y+=32
  @release.tween(self,[
     [Interpolator::ZOOM_X,0],
     [Interpolator::ZOOM_Y,0],
     [Interpolator::OPACITY,0]
  ],100)
  @startRelease=true
end
