def maxDigits
  if @maxDigits>0
    return @maxDigits
  else
    return [numDigits(self.minNumber),numDigits(self.maxNumber)].max
  end
end

def maxNumber
  ret=0
  if @maxDigits>0
    ret=((10**@maxDigits)-1)
  elsif
    ret=@maxNumber
  end
  ret=0 if !@negativeAllowed && ret<0
  return ret
end
