def refresh
  w=@movesel.width
  h=@movesel.height/2
  self.x=240
  self.y=92+(self.index*64)
  self.y-=76 if @fifthmove
  self.y+=20 if @fifthmove && self.index==4
  self.bitmap=@movesel.bitmap
  if self.preselected
    self.src_rect.set(0,h,w,h)
  else
    self.src_rect.set(0,0,w,h)
  end
end
